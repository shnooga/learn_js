Javascript class/struct with public properties and method
(efficient mem usage via prototype method and static property, PROPERTY_TAX)

function House(address, rooms, price, hasGarage) {
this.address = address;
this.rooms = rooms;
this.price = price;
this.hasGarage = hasGarage;
}
House.PROPERTY_TAX= .01;
House.prototype.view = function() {
document.write(this.address + ' has '+ this.rooms +' rooms, '+(this.hasGarage ? 'a' : 'no')+' garage, and costs $' + this.price);
}
House.prototype.totalCost = function() {
document.write('Total cost is: $' + eval(this.price + this.price * House.PROPERTY_TAX));
}
