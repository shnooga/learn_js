Rudimentary javascript class instantiaton



function House(address, rooms, price, garage) {
this.address=address;
this.rooms=rooms;
this.price=price;
this.garage=garage;

this.view = function() {
document.write(this.address + ' has '+ this.rooms+' rooms, '+(this.garage?'a':'no')+' garage, and costs $' + this.price);
}

this.totalCost = function() {
document.write('Total cost is: $' + eval(this.price + this.price * this.PROPERTY_TAX));
}
}
House.prototype.PROPERTY_TAX= .01;
