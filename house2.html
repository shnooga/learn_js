Javascript class with private properties and public accessors
(at the cost of inefficent mem usage regarding methods)



function House(address, rooms, price, garage) {
var _address = address;
var _rooms = rooms;
var _price = price;
var _garage = garage;

function getAddress() { return _address; }
function getRooms() { return _rooms; }
function getPrice() { return _price; }
function hasGarage() { return _garage; }

// Expose only these methods return {
getAddress: getAddress,
getRooms: getRooms,
getPrice: getPrice,
hasGarage: hasGarage,
}
}

function view(myHouse) {
document.write(myHouse.getAddress() + ' has '+ myHouse.getRooms() +' rooms, '+(myHouse.hasGarage() ? 'a' : 'no')+' garage, and costs $' + myHouse.getPrice());
}

function totalCost(myHouse) {
document.write('Total cost is: $' + eval(myHouse.getPrice() + myHouse.getPrice() * myHouse.PROPERTY_TAX));
}
