Rudimentary javascript class instantiaton



function House(address, rooms, price, garage) {
this.address=address;
this.rooms=rooms;
this.price=price;
this.garage=garage;
}

function view(aHouse) {
document.write(aHouse.address + ' has '+ aHouse.rooms+' rooms, '+(aHouse.garage?'a':'no')+' garage, and costs $' + aHouse.price);
}

function totalCost(aHouse) {
document.write('Total cost is: $' + eval(aHouse.price + aHouse.price * aHouse.PROPERTY_TAX));
}
House.prototype.PROPERTY_TAX= .01;
