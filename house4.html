Javascript class/struct with public properties and method
(efficient mem usage via prototype method and static property, PROPERTY_TAX)

function House(address, rooms, price, hasGarage) {
// Private member
var _address = address;
var _rooms = rooms;
var _price = price;
var _hasGarage = hasGarage;
// Private method
function priv_view() {
return _address + ' has '+ _rooms +' rooms, ' + (_hasGarage ? 'a' : 'no') + ' garage, and costs $' + _price;
}

// Privilege method
this.view = function() {
document.write(priv_view());
}

this.totalCost = function() {
document.write('Total cost is: $' + eval(_price + _price * House.PROPERTY_TAX));
}
}
House.PROPERTY_TAX= .01;
